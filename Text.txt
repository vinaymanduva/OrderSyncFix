Here’s the APEX code for the Service: 
public with sharing class OrderSyncHandler {
public static void syncOrder(Id orderId) {
Order__c o = [SELECT Id, Name, Status__c, Amount__c, Discount__c, External_Id__c,
(SELECT Id, ProductCode__c, Quantity__c, UnitPrice__c, LineAmount__c FROM OrderItems__r)
FROM Order__c
WHERE Id = :orderId
LIMIT 1];
if (o.Status__c != 'Activated') {
System.debug('Order not activated, skipping');
return;
}
// Build payload (note: discount field may be null)
Map<String, Object> payload = new Map<String, Object>{
'orderId' => (o.External_Id__c != null ? o.External_Id__c : String.valueOf(o.Id)),
'orderName' => o.Name,
'amount' => o.Amount__c,
'discount' => o.Discount__c,
'lines' => new List<Map<String, Object>>()
};
for (OrderItem__c i : o.OrderItems__r) {
Decimal lineAmt = (i.UnitPrice__c * i.Quantity__c) - (o.Discount__c == null ? 0 : (o.Discount__c / o.OrderItems__r.size()));
payload.get('lines').add(new Map<String, Object>{
'sku' => i.ProductCode__c,
'qty' => i.Quantity__c,
'unitPrice' => i.UnitPrice__c,
'amount' => lineAmt
});
}
Http h = new Http();
HttpRequest req = new HttpRequest();
req.setEndpoint(URL.getSalesforceBaseUrl().toExternalForm() + '/services/apexrest/OrderSyncAPI');
req.setMethod('POST');
req.setHeader('Content-Type','application/json');
req.setBody(JSON.serialize(payload));
try {
HttpResponse res = h.send(req);
if (res.getStatusCode() == 200) {
System.debug('Sync success: ' + res.getBody());
} else if (res.getStatusCode() == 202) {
System.debug('Accepted for async processing');
} else if (res.getStatusCode() == 409) {
System.debug('Duplicate, ignoring');
} else {
System.debug('Unexpected response: ' + res.getStatus() + ' body=' + res.getBody());
}
} catch (Exception e) {
System.debug('Error calling OrderSyncAPI: ' + e.getMessage());
}
}
}
